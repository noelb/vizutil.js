Find the intersection between two different line segments. 

// Two lines that don't cross, but would cross each other if extended
var line0 = [ {x:0,y:0}, {x:10,y:10} ] ;
var line1 = [ {x:100,y:0}, {x:90,y:10} ] ;

// FINDS THE INTERSECTION BETWEEN LINE SEGMENTS
// The example segments don't cross, so this returns null
// test0 = null
var test0 = GeomUtil.intersect(
    line0[0].x,line0[0].y,line0[1].x,line0[1].y, //First line
    line1[0].x,line1[0].y,line1[1].x,line1[1].y //Second line
    );

// FINDS THE INTERSECTION BETWEEN LINES
// Using lines (instead of segments) return an intersection
// test1 = [50,50]
var test1 = GeomUtil.intersect(
    line0[0].x,line0[0].y,line0[1].x,line0[1].y, //First line
    line1[0].x,line1[0].y,line1[1].x,line1[1].y, //Second line
    false);

// YOU CAN CHOOSE RESPONSE FORMAT

// Two lines that cross each other
var line0 = [ {x:0,y:0}, {x:100,y:100} ] ;
var line1 = [ {x:100,y:0}, {x:0,y:100} ] ;

// inter = [50,50]
var inter = GeomUtil.intersect(
    line0[0].x,line0[0].y,line0[1].x,line0[1].y, //First line
    line1[0].x,line1[0].y,line1[1].x,line1[1].y  //Second line
    );

// interXY = {x:50,y:50}
var interXY = GeomUtil.intersect(
